// Obecny kod ma ograniczenia - proponuję rozszerzenie: const translations = { pl: { "Polska fachowość, niemiecka precyzja": "Polska fachowość, niemiecka precyzja", "Zaufaj doświadczeniu...": "Zaufaj doświadczeniu i rzetelności...", // wszystkie teksty po polsku }, de: { "Polska fachowość, niemiecka precyzja": "Polnische Facharbeit, deutsche Präzision", "Zaufaj doświadczeniu...": "Vertrauen Sie auf Erfahrung...", // wszystkie teksty po niemiecku } }; function setLang(lang) { document.documentElement.lang = lang; document.querySelectorAll('[data-pl], [data-de]').forEach(el => { const key = el.getAttribute('data-pl') || el.textContent; if (translations[lang] && translations[lang][key]) { el.textContent = translations[lang][key]; } }); // Zapisz preferencje użytkownika localStorage.setItem('preferredLang', lang); } // Przy ładowaniu strony document.addEventListener('DOMContentLoaded', function() { const savedLang = localStorage.getItem('preferredLang') || 'pl'; setLang(savedLang); });
